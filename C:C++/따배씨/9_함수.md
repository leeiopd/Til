```c
#include <stdio.h>
#include <string.h>
#include <stdbool.h>

#define WIDTH 20
#define NAME "LEE"
#define ADDRESS "Seoul, Korea"

void print_multiple_chars(char c, int n_stars, bool endl);
void print_centered_string(char str[]);

int main(){

    print_multiple_chars('*', WIDTH, true);
    print_centered_string(NAME);
    print_centered_string(ADDRESS);
    print_multiple_chars('*', WIDTH, true);
    
    
    return 0;
}

void print_multiple_chars(char c, int n, bool endl){
    
    for (int i = 0; i < n; ++i){
        printf("%c", c);
    }
    
    if (endl)
        printf("\n");
}

void print_centered_string(char str[])
{
    int n_blanks = 0;
    n_blanks = (int)(WIDTH - strlen(str)) / 2;
    print_multiple_chars(' ', n_blanks, false);
    printf("%s\n", str);
}
```
